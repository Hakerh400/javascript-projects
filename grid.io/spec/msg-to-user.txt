00 - Grid fragment
  uint16le - X
  uint16le - Y
  uint16le - Width
  uint16le - Height
  n bytes - Tiles
    uint16le - Type
01 - Single tile
  uint16le - X
  uint16le - Y
  uint16le - Type
02 - New entity
  uint64le - Entity ID
  float64le - X
  float64le - Y
  float32le - Direction
  uint16le - Type
  n bytes - Extra data
03 - Update entity
  uint64le - Entity ID
  uint8 - Type of the update
    bit 0 - Coordinates
      float64le - X
      float64le - Y
    bit 1 - Direction
      float32le - Direction
    bit 2 - Extra data
      n bytes = Extra data
04 - Remove entity
  uint64le - Entity ID
  uint8 - Reason
    00 - Outside of the viewport
    01 - Dead
