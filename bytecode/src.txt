class Main{
  Main(){}

  Object main(Object args){
    String str = args;

    Array<String> arr = str.split(" ");
    Array<Integer> intArr = arr.map(this.mapFunc);

    Integer initVal = 0;
    Integer result = intArr.reduce(this.reduceFunc, initVal);

    return result;
  }

  Integer mapFunc(String str, Integer index){
    Integer result = str.parseInt(10);
    return result;
  }

  Integer reduceFunc(Integer prev, Integer curr, Integer index){
    Integer result = prev.add(curr);
    return result;
  }
}
